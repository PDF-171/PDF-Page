let currentPage = 1; document.addEventListener('DOMContentLoaded', () => { const flyerImage = document.getElementById('flyer-image'); const prevBtn = document.getElementById('prevBtn'); const nextBtn = document.getElementById('nextBtn'); const downloadBtn = document.getElementById('downloadBtn'); prevBtn.addEventListener('click', () => showPage(1)); nextBtn.addEventListener('click', () => showPage(2)); downloadBtn.addEventListener('click', downloadPDF); updateButtons(); function showPage(page) { if (page === 1 && currentPage !== 1) { flyerImage.src = 'apollo-flyer-page1.jpg'; currentPage = 1; updateButtons(); } else if (page === 2 && currentPage !== 2) { flyerImage.src = 'apollo-flyer-page2.jpg'; currentPage = 2; updateButtons(); } } function updateButtons() { if (currentPage === 1) { prevBtn.classList.remove('active'); prevBtn.disabled = true; nextBtn.classList.add('active'); nextBtn.disabled = false; } else { prevBtn.classList.add('active'); prevBtn.disabled = false; nextBtn.classList.remove('active'); nextBtn.disabled = true; } } function downloadPDF() { const a = document.createElement('a'); a.href = 'apollo-flyer.pdf'; a.download = 'apollo-flyer.pdf'; a.click(); } });
